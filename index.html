1. Create a New User on the Server

sudo adduser your_username
sudo usermod -aG sudo your_username
sudo mkdir -p /home/your_username/.ssh
sudo chown your_username:your_username /home/your_username/.ssh
sudo chmod 700 /home/your_username/.ssh

2. Generate SSH Key from Windows to Server

ssh-keygen -t ed25519 -C "key-local-to-server" -f $env:USERPROFILE\\.ssh\\id_yourkey
Get-Content $env:USERPROFILE\\.ssh\\id_yourkey.pub | Set-Clipboard
sudo nano /home/your_username/.ssh/authorized_keys
sudo chmod 600 /home/your_username/.ssh/authorized_keys
sudo chown your_username:your_username /home/your_username/.ssh/authorized_keys
ssh -i $env:USERPROFILE\\.ssh\\id_yourkey -p 13976 your_username@your.server.ip

3. Generate GitHub Key from Server

ssh-keygen -t ed25519 -C "github-key" -f ~/.ssh/id_github
cat ~/.ssh/id_github.pub
ssh -T git@github.com

4. Clone the Project from GitHub

cd ~
git clone git@github.com:YourUsername/your-repo.git
cd your-repo

5. Setup Python Environment

python3 -m venv venv
source venv/bin/activate
pip install --upgrade pip
pip install -r requirements.txt

6. Run Flask Locally

export FLASK_APP=app.py
export FLASK_ENV=development
flask run --host=0.0.0.0 --port=5000

7. Setup Gunicorn and Nginx

gunicorn --bind 127.0.0.1:8000 myapp:app
server {
    listen 80;
    server_name yourdomain.com;

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
sudo ln -s /etc/nginx/sites-available/your_username /etc/nginx/sites-enabled/
sudo nginx -t
sudo systemctl reload nginx

8. Enable HTTPS with Certbot

sudo apt install certbot python3-certbot-nginx -y
sudo certbot --nginx -d yourdomain.com --http-01-port 80

9. Enable UFW Firewall

sudo ufw allow OpenSSH
sudo ufw allow 80,443,5000/tcp
sudo ufw enable
